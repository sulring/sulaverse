4import re
5from dataclasses import dataclass, field
6from pathlib import Path

9# Matches: --name: hsl(Ndeg, N%, N%)
10HSL_RE = re.compile(
11 r"--(?P<name>[\w-]+)\s*:\s*hsl\((?P<h>[\d.]+)deg,\s*(?P<s>[\d.]+)%,\s*(?P<l>[\d.]+)%\)"
12)

14@dataclass(frozen=True)
15class Color:
16 """Immutable HSL color with format converters."""
17 name: str
18 h: int
19 s: float
20 l: float
21
22 def to_rgb(self) -> tuple[int, int, int]:
23 s, l = self.s / 100, self.l / 100
24 a = s * min(l, 1 - l)
25 def f(n: int) -> int:
26 k = (n + self.h / 30) % 12
27 return round((l - a * max(-1, min(k - 3, min(9 - k, 1)))) * 255)
28 return f(0), f(8), f(4)
29
30 def to_hex(self) -> str:
31 return "#" + "{:02x}{:02x}{:02x}".format(*self.to_rgb())
32
33 def to_hsl(self) -> str:
34 return f"hsl({self.h}deg, {self.s:.0f}%, {self.l:.0f}%)"
35
36def iter_colors(css: str) -> Iterator[Color]:
37 """Yield Color objects from raw CSS text."""
38 for m in HSL_RE.finditer(css):
39 d = m.groupdict()
40 yield Color(d["name"], int(d["h"]), float(d["s"]), float(d["l"]))
41
42def load_palette(path: str | Path = "color-theme-extended.css") -> dict[str, Color]:
43 """Read CSS file and return mapping of token → Color."""
44 css = Path(path).read_text()
45 return {c.name: c for c in iter_colors(css)}
